def schedule_review(user, topic, performance_score):
# performance_score: 0 to 100 (e.g., test score)
text
    # Using a simplified version of the SuperMemo algorithm
    # We'll adjust the interval and ease factor based on performance

    try:
        schedule = SpacedRepetitionSchedule.objects.get(user=user, topic=topic)
    except SpacedRepetitionSchedule.DoesNotExist:
        schedule = SpacedRepetitionSchedule(user=user, topic=topic)

    if performance_score >= 90:
        # Good performance, increase interval
        schedule.interval = schedule.interval * schedule.ease_factor
        schedule.ease_factor = max(1.3, schedule.ease_factor + 0.1)
    else:
        # Poor performance, reset interval
        schedule.interval = 1
        schedule.ease_factor = max(1.3, schedule.ease_factor - 0.2)

    schedule.repetitions += 1
    schedule.next_review_date = timezone.now() + timedelta(days=schedule.interval)
    schedule.save()
